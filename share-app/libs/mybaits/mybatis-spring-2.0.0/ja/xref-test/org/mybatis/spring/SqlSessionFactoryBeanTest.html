146     customConfiguration.setUseGeneratedKeys(true);
147     customConfiguration.setDefaultExecutorType(ExecutorType.REUSE);

149     factoryBean.setConfiguration(customConfiguration);

153     assertThat(factory.getConfiguration().getEnvironment().getId()).isEqualTo(SqlSessionFactoryBean.class.getSimpleName());
154     assertThat(factory.getConfiguration().getEnvironment().getDataSource()).isSameAs(dataSource);
155     assertThat(factory.getConfiguration().getEnvironment().getTransactionFactory().getClass()).isSameAs(SpringManagedTransactionFactory.class);
156     assertThat(factory.getConfiguration().getVfsImpl()).isSameAs(JBoss6VFS.class);

158     assertThat(factory.getConfiguration().isCacheEnabled()).isFalse();
159     assertThat(factory.getConfiguration().isUseGeneratedKeys()).isTrue();
160     assertThat(factory.getConfiguration().getDefaultExecutorType()).isSameAs(ExecutorType.REUSE);
161   }

247   @Test
248   void testSetConfigLocation() throws Exception {
249     setupFactoryBean();
250 
251     factoryBean.setConfigLocation(new org.springframework.core.io.ClassPathResource(
252         "org/mybatis/spring/mybatis-config.xml"));
253 
254     SqlSessionFactory factory = factoryBean.getObject();
255 
256     assertThat(factory.getConfiguration().getEnvironment().getId()).isEqualTo(SqlSessionFactoryBean.class.getSimpleName());
257     assertThat(factory.getConfiguration().getEnvironment().getDataSource()).isSameAs(dataSource);
258     assertThat(factory.getConfiguration().getEnvironment().getTransactionFactory().getClass())
259         .isSameAs(org.mybatis.spring.transaction.SpringManagedTransactionFactory.class);
260     assertThat(factory.getConfiguration().getVfsImpl()).isSameAs(JBoss6VFS.class);
261 
262     // properties explicitly set differently than the defaults in the config xml
263     assertThat(factory.getConfiguration().isCacheEnabled()).isFalse();
264     assertThat(factory.getConfiguration().isUseGeneratedKeys()).isTrue();
265     assertThat(factory.getConfiguration().getDefaultExecutorType()).isSameAs(org.apache.ibatis.session.ExecutorType.REUSE);
266 
267     // for each statement in the xml file: org.mybatis.spring.TestMapper.xxx & xxx
268     assertThat(factory.getConfiguration().getMappedStatementNames().size()).isEqualTo(8);
269 
270     assertThat(factory.getConfiguration().getResultMapNames().size()).isEqualTo(0);
271     assertThat(factory.getConfiguration().getParameterMapNames().size()).isEqualTo(0);
272   }

403   private void assertDefaultConfig(SqlSessionFactory factory) {
404     assertConfig(factory, SqlSessionFactoryBean.class.getSimpleName(),
405         org.mybatis.spring.transaction.SpringManagedTransactionFactory.class);
406     assertThat(factory.getConfiguration().getVariables().size()).isEqualTo(0);
407   }
408 
409   private void assertConfig(SqlSessionFactory factory, Class<? extends TransactionFactory> transactionFactoryClass) {
410     assertConfig(factory, SqlSessionFactoryBean.class.getSimpleName(), transactionFactoryClass);
411   }
412 
413   private void assertConfig(SqlSessionFactory factory, String environment,
414       Class<? extends TransactionFactory> transactionFactoryClass) {
415     assertThat(factory.getConfiguration().getEnvironment().getId()).isEqualTo(environment);
416     assertThat(factory.getConfiguration().getEnvironment().getDataSource()).isSameAs(dataSource);
417     assertThat(factory.getConfiguration().getEnvironment().getTransactionFactory().getClass()).isSameAs(transactionFactoryClass);
418 
419     // no mappers configured => no mapped statements or other parsed elements
420     assertThat(factory.getConfiguration().getMappedStatementNames().size()).isEqualTo(0);
421     assertThat(factory.getConfiguration().getResultMapNames().size()).isEqualTo(0);
422     assertThat(factory.getConfiguration().getParameterMapNames().size()).isEqualTo(0);
423     assertThat(factory.getConfiguration().getSqlFragments().size()).isEqualTo(0);
424   }
